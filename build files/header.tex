% Default image sizing: cap figures at text width / 45% of page height and
% center them on the line. The 0.55 shrink is applied by patching the
% Pandoc \pandocbounded macro below.
\adjustboxset{max width=\textwidth, max height=0.45\textheight, keepaspectratio, center}

% Force figures to stay inline with the text rather than floating to their own
% pages.  Pandoc's default template sets \fps@figure to htbp, so we override it
% after the preamble is complete using the float package's placement command.
\AtBeginDocument{%
  \floatplacement{figure}{H}%
  \makeatletter
  \gdef\fps@figure{H}%
  \makeatother
  % Shrink Pandoc figures by ~45% so they sit inline with the text.
  % Also force each figure onto its own centered line so text does not wrap on the same line.
  \let\oldpandocbounded\pandocbounded
  \renewcommand{\pandocbounded}[1]{\par{\centering\scalebox{0.55}{\oldpandocbounded{#1}}\par}}%
}
